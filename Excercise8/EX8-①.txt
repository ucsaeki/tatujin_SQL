EX8-① 結合が先か、集約が先か？

/*年齢、性別、地方ごとの人口を表示する*/

SELECT
	MASTER.age_class,
	MASTER.sex_cd,
	SUM(
		CASE
			WHEN pref_name IN ('青森','秋田')THEN population
			ELSE NULL
		END
		) AS pop_tohoku,
	SUM(
		CASE
			WHEN pref_name IN ('東京','千葉')THEN population
			ELSE NULL
		END
		) AS pop_kanto
FROM
	(
		SELECT
			age_class,
			sex_cd
		FROM
			ages
		CROSS JOIN
			tbl_sexes
	) AS MASTER
LEFT OUTER JOIN
	tbl_pops AS t
ON
	t.age_class = MASTER.age_class
AND
	t.sex_cd = MASTER.sex_cd
GROUP BY
	age_class,
	sex_cd DESC
;
