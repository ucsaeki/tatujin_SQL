EX1-① 複数列の最大値


・xとyの最大値

SELECT
	key_code AS 'key',
	CASE 
		WHEN x > y THEN x
		ELSE y
	END AS 'greatests'
FROM
	greatests
;



・x,y,zの最大値

SELECT
	key_code AS 'key',
	CASE
		WHEN x > y THEN
		(
			CASE
				WHEN x > z THEN x 
				ELSE z
			END
		)
		WHEN y > x THEN
		(
			CASE
				WHEN y > z THEN y
				ELSE z
			END
		)
	ELSE 
		(
			CASE
				WHEN x > z THEN x
				ELSE z
			END
		)
	END AS 'greatests'
FROM
	greatests
;
