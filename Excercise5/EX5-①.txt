EX5-① 配列テーブル――行持ちの場合


・EXISTS使用

SELECT DISTINCT
	number_key
FROM
	Array_Tables2 AS AT1
WHERE EXISTS
	(
		SELECT
			number_key
		FROM
			Array_Tables2 AS AT2
		WHERE
			AT2.number_key = AT1.number_key
		AND
			/*1は問題で指定されたエンティティを示す値*/
			val = 1
	)
;


・EXISTS不使用

SELECT DISTINCT
	number_key
FROM
	Array_Tables2
WHERE
	val = 1
;
