EX9-① UNIONだけを使うコンペアの改良


/*二つのテーブルの相当性を比較する*/

SELECT
	(
		SELECT
			COUNT(*)
		FROM
			a_codes
	) AS 'COUNT(a_codes)',
	(
		SELECT
			COUNT(*)
		FROM
			b_codes
	) AS 'COUNT(b_codes)',
	COUNT(*) AS row_cnt
FROM
	(
		SELECT
			*
		FROM
			a_codes
	UNION
		SELECT
			*
		FROM
			b_codes
	) AS TMP
;
